import pandas as pd
import numpy as np
import plotly.graph_objects as go
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error
from datetime import timedelta

def calcular_rsi(data, window=14):
    delta = data.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    rs = gain / (loss + 1e-9)
    return 100 - (100 / (1 + rs))

def backtest_anual_3_em_3(file_path):
    # 1. Carregamento
    df_base = pd.read_csv(file_path)
    df_base['Data'] = pd.to_datetime(df_base['Data'])
    df_base = df_base.sort_values('Data').reset_index(drop=True)

    # 2. Criação de Features
    df = df_base.copy()
    for i in range(1, 11):
        df[f'lag_{i}'] = df['Último'].shift(i)
    
    df['ma_7'] = df['Último'].rolling(window=7).mean()
    df['rsi'] = calcular_rsi(df['Último'])
    df['impulso'] = df['Último'].diff() 
    df = df.dropna().reset_index(drop=True)

    features = [f'lag_{i}' for i in range(1, 11)] + ['ma_7', 'rsi', 'impulso']

    # --- LÓGICA DE BACKTEST ANUAL ---
    dias_totais = 70
    passo = 7
    indice_inicio = len(df) - dias_totais
    
    todas_previsoes = []
    todos_reais = []
    todas_datas = []
    # Novas listas para o Candlestick
    todas_aberturas = []
    todas_maximas = []
    todas_minimas = []

    print(f"Iniciando Backtest Anual (Simulando {dias_totais} dias)...")

    for i in range(indice_inicio, len(df) - passo, passo):
        df_train = df.iloc[:i].copy()
        df_test_real = df.iloc[i:i+passo].copy()
        
        last_train_price = df_train.iloc[-1]['Último']
        last_train_features = df_train.tail(1)[features]
        current_date = df_train.iloc[-1]['Data'].date()

        print(f"Processando a partir de: {current_date}...")

        preds_janela = []
        for h in range(1, passo + 1):
            y_trend = df_train['Último'].shift(-h) - df_train['Último']
            y_trend = y_trend.dropna()
            X_trend = df_train[features].iloc[:len(y_trend)]
            
            model = RandomForestRegressor(
                n_estimators=2800, 
                max_features='sqrt',
                min_samples_leaf=5,
                random_state=900, 
                n_jobs=-1
            )
            model.fit(X_trend, y_trend)
            
            pred_var = model.predict(last_train_features)[0]
            preds_janela.append(last_train_price + pred_var)

        todas_previsoes.extend(preds_janela)
        todos_reais.extend(df_test_real['Último'].tolist())
        todas_datas.extend(df_test_real['Data'].tolist())
        # Capturando dados OHLC para as velas
        todas_aberturas.extend(df_test_real['Abertura'].tolist())
        todas_maximas.extend(df_test_real['Máxima'].tolist())
        todas_minimas.extend(df_test_real['Mínima'].tolist())

    # 5. Métricas e Gráfico Final
    mae = mean_absolute_error(todos_reais, todas_previsoes)
    print(f"\n--- RESULTADO FINAL DO BACKTEST ANUAL ---")
    print(f"Erro Médio Absoluto (MAE) no Ano: {mae:.5f}")

    fig = go.Figure()

    # Adicionando Velas (Candlesticks)
    fig.add_trace(go.Candlestick(
        x=todas_datas,
        open=todas_aberturas,
        high=todas_maximas,
        low=todas_minimas,
        close=todos_reais,
        name='Preço Real (Velas)'
    ))

    # Mantendo a linha de Previsão da IA
    fig.add_trace(go.Scatter(
        x=todas_datas, y=todas_previsoes,
        mode='lines', name='Previsão da IA',
        line=dict(color='orange', width=2, dash='dash')
    ))

    fig.update_layout(
        title=f'Backtest Gold - Real (Candlestick) vs Previsão IA (MAE: {mae:.2f})',
        template='plotly_dark',
        xaxis_title='Data', 
        yaxis_title='Preço',
        xaxis_rangeslider_visible=False # Remove a barra de rolagem inferior para foco total
    )
    
    fig.show()

# Executar
backtest_anual_3_em_3('XAU_USD Historical Data_LIMPO.csv')
